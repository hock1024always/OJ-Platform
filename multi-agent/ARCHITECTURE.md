### 6. йҖҡдҝЎеҚҸи®®

```go
// Agent й—ҙж¶ҲжҒҜ
type AgentMessage struct {
    MessageID   string                 `json:"message_id"`
    From        string                 `json:"from"`        // Agent ID
    To          string                 `json:"to"`          // Agent ID жҲ– broadcast
    Type        MessageType            `json:"type"`        // task/request/response/event
    TaskID      string                 `json:"task_id"`     // е…іиҒ”д»»еҠЎ
    Content     map[string]interface{} `json:"content"`
    Timestamp   int64                  `json:"timestamp"`
}

type MessageType string

const (
    MessageTypeTask     MessageType = "task"     // еҲҶй…Қд»»еҠЎ
    MessageTypeRequest  MessageType = "request"  // иҜ·жұӮеҚҸеҠ©
    MessageTypeResponse MessageType = "response" // е“Қеә”
    MessageTypeEvent    MessageType = "event"    // зҠ¶жҖҒдәӢд»¶
    MessageTypeResult   MessageType = "result"   // жңҖз»Ҳз»“жһң
)
```
